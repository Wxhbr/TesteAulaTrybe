<!DOCTYPE html>
<html>
    <head>
        <meta charset="utf-8">
        <title>Challenge: Seasonal ids</title>
        <style>
            
            #summer {
                color: rgb(219, 219, 51);
                background-color: rgb(189, 32, 32);
            }

                color: rgb(9, 21, 153);

    <body>

        <p>He likes to stay in and cuddle</p>
    </body>
